Tech Stack (Final Decisions)
Frontend: Next.js 14 with App Router + TypeScript + Tailwind CSS

Why Next.js over plain React: Server components for SEO, API routes as BFF, better performance
Why TypeScript: You will thank yourself in month 2 when the data structures get complex

Backend: Python FastAPI

Why Python over Node: NumPy/Pandas for financial calculations, pdfplumber for PDF parsing, vastly better data ecosystem
Why FastAPI over Flask/Django: Async support, auto-generated API docs, type safety with Pydantic

Database: Supabase (PostgreSQL, free tier)

Why not MongoDB: Financial data is highly relational. Fund → Holdings → Stocks. Use relational DB.

Cache: Upstash Redis (free tier: 10k requests/day)
Charts: Recharts (free, React-native, good enough)

Upgrade path: D3.js when you need custom overlap heatmaps

LLM: Google Gemini 1.5 Flash (free tier: 1M tokens/day)
PDF Parsing: pdfplumber (Python, free)
Deployment:

Frontend: Vercel (free)
Backend: Railway (free tier: $5 credit/month, enough for hobby)
DB: Supabase (free tier)
Cache: Upstash (free tier)

-- ============================================
-- FUNDS MASTER (populated once from AMFI)
-- ============================================
CREATE TABLE funds (
  scheme_code        VARCHAR(10) PRIMARY KEY,
  scheme_name        TEXT NOT NULL,
  amc_name           TEXT NOT NULL,
  category           TEXT,          -- 'Large Cap', 'Flexi Cap', etc.
  sub_category       TEXT,
  plan_type          TEXT,          -- 'Direct', 'Regular'
  option_type        TEXT,          -- 'Growth', 'IDCW'
  nav                DECIMAL(12,4),
  nav_date           DATE,
  expense_ratio      DECIMAL(5,3),  -- in percentage
  aum_crore          DECIMAL(15,2),
  launch_date        DATE,
  is_active          BOOLEAN DEFAULT true,
  created_at         TIMESTAMP DEFAULT NOW(),
  updated_at         TIMESTAMP DEFAULT NOW()
);

-- ============================================
-- FUND HOLDINGS (from AMFI monthly disclosures)
-- ============================================
CREATE TABLE fund_holdings (
  id                 SERIAL PRIMARY KEY,
  scheme_code        VARCHAR(10) REFERENCES funds(scheme_code),
  disclosure_date    DATE NOT NULL,
  stock_isin         VARCHAR(20),
  stock_name         TEXT NOT NULL,
  sector             TEXT,
  holding_percentage DECIMAL(6,3),  -- % of fund's AUM
  market_value_cr    DECIMAL(15,2),
  asset_type         TEXT,          -- 'Equity', 'Debt', 'Cash', 'Other'
  created_at         TIMESTAMP DEFAULT NOW(),
  UNIQUE(scheme_code, disclosure_date, stock_isin)
);

-- ============================================
-- USER SESSIONS (no auth in MVP)
-- ============================================
CREATE TABLE analysis_sessions (
  session_id         UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  input_data         JSONB NOT NULL,   -- list of {fund_name, units, purchase_date}
  analysis_result    JSONB,            -- complete analysis output
  health_score       SMALLINT,         -- 0-100
  created_at         TIMESTAMP DEFAULT NOW(),
  expires_at         TIMESTAMP DEFAULT (NOW() + INTERVAL '7 days')
);

-- ============================================
-- FOR PHASE 2: USER AUTH + PORTFOLIOS
-- ============================================
CREATE TABLE users (
  user_id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  email              TEXT UNIQUE NOT NULL,
  name               TEXT,
  created_at         TIMESTAMP DEFAULT NOW()
);

CREATE TABLE portfolios (
  portfolio_id       UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id            UUID REFERENCES users(user_id),
  name               TEXT DEFAULT 'My Portfolio',
  holdings           JSONB NOT NULL,
  last_analyzed      TIMESTAMP,
  created_at         TIMESTAMP DEFAULT NOW()
);

-- ============================================
-- INDEXES (critical for performance)
-- ============================================
CREATE INDEX idx_fund_holdings_scheme ON fund_holdings(scheme_code);
CREATE INDEX idx_fund_holdings_date ON fund_holdings(disclosure_date);
CREATE INDEX idx_fund_holdings_isin ON fund_holdings(stock_isin);
CREATE INDEX idx_funds_amc ON funds(amc_name);
CREATE INDEX idx_funds_category ON funds(category);
